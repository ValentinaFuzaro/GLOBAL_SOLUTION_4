CREATE TABLE USUARIO_MEDICO (
	ID_MEDICO 					 NUMBER(3) NOT NULL,
    NM_MEDICO                    VARCHAR2(100) NOT NULL,
    TIPO_ESPECIALIDADE           VARCHAR2(50) NOT NULL,
    CRM                          VARCHAR2(20) NOT NULL,
    NR_CPF                       VARCHAR2(11) NOT NULL,
    DT_NASCIMENTO                DATE,
    SEXO                         CHAR(1),
    NM_ENDERECO                  VARCHAR2(255) NOT NULL,
